38 AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE Index
39 bisection_search(const It begin, const It end, const T& x)
40 {
41  auto idx = check_bounds(begin, end, x);
42  if ((idx.lim == Limits::LOWLIM) || (idx.lim == Limits::UPLIM)) {
43  return idx;
44  }
45

49

58  }

83 AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
84  typename std::iterator_traits<C2>::value_type
85  linear(
86  const C1 xbegin,
87  const C1 xend,
88  const C2 yinp,
89  const typename std::iterator_traits<C1>::value_type& xout)
90 {

93

95  return yinp[idx.idx];

98  const int j = idx.idx;
99  const auto denom = (xbegin[j + 1] - xbegin[j]);

102  return facL * yinp[j] + facR * yinp[j + 1];

134  yout[i] = facL * yinp[j] + facR * yinp[j + 1];

156 // Angles in yinp should be defined between [-upper_bound, +upper_bound]

158 template <typename C1, typename C2>
159 AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
160  typename std::iterator_traits<C2>::value_type
161  linear_angle(
162  const C1 xbegin,
163  const C1 xend,
164  const C2 yinp,
165  const typename std::iterator_traits<C1>::value_type& xout,
166  const typename std::iterator_traits<C1>::value_type& upper_bound)
167 {
168  using DType1 = typename std::iterator_traits<C1>::value_type;
169  const auto idx = bisection_search(xbegin, xend, xout);
170 
171  if ((idx.lim == Limits::LOWLIM) || (idx.lim == Limits::UPLIM)) {
172  return yinp[idx.idx];
173  }
174  static constexpr DType1 eps = 1.0e-8;
175  const int j = idx.idx;
176  const auto denom = (xbegin[j + 1] - xbegin[j]);
177  /* https://math.stackexchange.com/questions/2144234/interpolating-between-2-angles
178  */
179  const auto R = (denom > eps) ? ((xout - xbegin[j]) / denom) : 0.0;
180  const auto ub = upper_bound;
181  const auto hb = 0.5 * upper_bound;
182  const auto ohb = 1.5 * upper_bound;
183  return yinp[j] +
184  (std::fmod((std::fmod(yinp[j + 1] - yinp[j], ub) + ohb), ub) - hb) *
185  R;
186 }
