442 template <typename FType>
443 inline void normalize(FType& field)
444 {
445  const amrex::Real eps = 1.0e-12;
446  const auto& repo = field.repo();
447  const auto ncomp = field.num_comp();
448 
449  const int nlevels = repo.num_active_levels();
450  for (int lev = 0; lev < nlevels; ++lev) {
451  const auto& farrs = field(lev).arrays();
452  amrex::ParallelFor(
453  field(lev),
454  [=] AMREX_GPU_DEVICE(int nbx, int i, int j, int k) noexcept {
455  auto farr = farrs[nbx];
456  // Compute magnitude
457  amrex::Real mag = 0.;
458  for (int icomp = 0; icomp < ncomp; ++icomp) {
459  mag = mag + farr(i, j, k, icomp) * farr(i, j, k, icomp);
460  }
461  if (mag > eps) {
462  for (int icomp = 0; icomp < ncomp; ++icomp) {
463  farr(i, j, k, icomp) =
464  farr(i, j, k, icomp) / std::sqrt(mag);
465  }
466  }
467  });

470 }
