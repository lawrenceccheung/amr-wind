37  double_data = 1;

43  // If double reading is deemed necessary, check for convenience
44  if (double_data == 1 && std::abs(wtime - newtime) <= 1e-10) {
45  // Reading can be done just once, w2a fields replace ow fields
46  double_data = 2;
47  }
48  }
49  // Check if reading must take place for other reasons
50  if (t_last < -1e-10) {
51  // Signifies initialization from scratch without waves or a restart
52  read_flag = true;
53  // Resizing needs to happen for the first time
54  resize_flag = true;
55 
56  // Confirm that new time is not coincident with modes time step
57  if (std::abs(wtime - newtime) > 1e-10) {
58  // Data must be read before and after wtime
59  double_data = 1;
60  } else {
61  // Data need only be read once, w2a fields will replace ow fields
62  double_data = 2;
63  }
64 
65  } else if (std::abs(t_last) < 1e-10) {
66  // Signifies initialization with waves

69  resize_flag = true;

539  // Read HOS data if necessary based on time
540  if (read_flag) {
541 
542  if (wdata.resize_flag) {
543  // Reset flag
544  wdata.resize_flag = false;
545  // Flags for indicating overlap, assume none at first
546  bool flag_z = false;
547  bool flag_xlo = false;
548  bool flag_xhi = false;
549  // Get heights for this processor, check overlap in z
550  flag_z =
551  (interp_to_mfab::get_local_height_indices(
552  wdata.indvec, wdata.hvec, m_ow_velocity.vec_ptrs(),
553  geom) == 1);
554  // No overlap from heights definitely means no interp
555 
556  // Check lateral bounds (in x)
557  const int dir = 0;
558  flag_xlo =
559  (interp_to_mfab::check_lateral_overlap_lo(
560  wdata.gen_length, dir, m_ow_velocity.vec_ptrs(),
561  geom) == 1);
562  // No overlap with gen region means no interp, unless ...
563  if (wdata.has_outprofile) {
564  // ... if overlap exists here, needing interp
565  flag_xhi =
566  (interp_to_mfab::check_lateral_overlap_hi(
567  wdata.beach_length, dir, m_ow_velocity.vec_ptrs(),
568  geom) == 1);
569  }
570 
571  if (flag_z && (flag_xlo || flag_xhi)) {
572  // Interpolation is needed
573  wdata.do_interp = true;
574  // Do resizing
575  wdata.sp_eta_vec.resize(
576  static_cast<size_t>(wdata.n0) *
577  static_cast<size_t>(wdata.n1),
578  0.0);
579  wdata.sp_u_vec.resize(
580  static_cast<size_t>(wdata.n0 * wdata.n1) *
581  wdata.indvec.size());
582  wdata.sp_v_vec.resize(
583  static_cast<size_t>(wdata.n0 * wdata.n1) *
584  wdata.indvec.size());
585  wdata.sp_w_vec.resize(
586  static_cast<size_t>(wdata.n0 * wdata.n1) *
587  wdata.indvec.size());
588  // Sizes will remain constant and need for interpolation
589  // will remain until a regrid occurs
590  } else {
591  // No overlapping with spatial data or no overlapping with
592  // relaxation zones, interpolation can be skipped
593  wdata.do_interp = false;
594  }
595  }
596  // Only perform reading where needed, communicate offset though
597  amrex::ParallelDescriptor::ReduceIntMax(wdata.n_offset);
598 
599  // If double read is required, then copy older wave data to ow_
600  // fields and modify interpolation parameters to get things right
601  if (double_data == 1) {
602  if (wdata.do_interp) {
603  populate_fields_all_levels(
604  wdata, geom, m_ow_levelset, m_ow_velocity, -1);
605  }
606 
607  // Average down to get fine information on coarse grid where
608  // possible (may be unnecessary)
609  for (int lev = nlevels - 1; lev > 0; --lev) {
610  amrex::average_down(
611  m_ow_velocity(lev), m_ow_velocity(lev - 1), 0,
612  AMREX_SPACEDIM, sim.mesh().refRatio(lev - 1));
613  amrex::average_down(
614  m_ow_levelset(lev), m_ow_levelset(lev - 1), 0, 1,
615  sim.mesh().refRatio(lev - 1));
616  }
617  // Fill patch to get correct ghost cells after average down
618  m_ow_velocity.fillpatch(sim.time().new_time());
619  m_ow_levelset.fillpatch(sim.time().new_time());
620 
621  // Prior t_last (corresponding to ow fields)
622  t_last = (wdata.ntime - 1) * wdata.dt_modes;
623  } else if (double_data == 2) {
624  // Restarting simulation or taking a big step, new time at ntime
625  // Initialize ow fields to 0 for time interp, will be replaced
626  m_ow_levelset.setVal(0.0);
627  m_ow_velocity.setVal(0.0);
628  // No modification needed for t_last, leads to interp factor = 1
629  }
630 
631  // After possible prior read, now read data for this ntime
632  if (wdata.do_interp) {
633  populate_fields_all_levels(
634  wdata, geom, w2a_levelset, w2a_velocity);
635  }
636 
637  // Average down to get fine information on coarse grid where
638  // possible (may be unnecessary)
639  for (int lev = nlevels - 1; lev > 0; --lev) {
640  amrex::average_down(
641  w2a_velocity(lev), w2a_velocity(lev - 1), 0, AMREX_SPACEDIM,
642  sim.mesh().refRatio(lev - 1));
643  amrex::average_down(
644  w2a_levelset(lev), w2a_levelset(lev - 1), 0, 1,
645  sim.mesh().refRatio(lev - 1));
646  }
647  // Fill patch to get correct ghost cells after average down
648  w2a_velocity.fillpatch(sim.time().new_time());
649  w2a_levelset.fillpatch(sim.time().new_time());
650  }
651 
652  // Temporally interpolate at every timestep to get target solution
653  for (int lev = 0; lev < nlevels; ++lev) {
654  for (amrex::MFIter mfi(m_ow_levelset(lev)); mfi.isValid(); ++mfi) {
655  auto phi = m_ow_levelset(lev).array(mfi);
656  auto vel = m_ow_velocity(lev).array(mfi);
657  auto W2A_phi = w2a_levelset(lev).array(mfi);
658  auto W2A_vel = w2a_velocity(lev).array(mfi);
659 
660  const amrex::Real W2A_t = wdata.t;
661  const auto& gbx = mfi.growntilebox(3);
662  amrex::ParallelFor(
663  gbx, [=] AMREX_GPU_DEVICE(int i, int j, int k) noexcept {
664  // Interpolate temporally every time
665  phi(i, j, k) += (W2A_phi(i, j, k) - phi(i, j, k)) *
666  (newtime - t_last) /
667  (W2A_t - t_last + 1e-16);
668  vel(i, j, k, 0) +=
669  (W2A_vel(i, j, k, 0) - vel(i, j, k, 0)) *
670  (newtime - t_last) / (W2A_t - t_last + 1e-16);
671  vel(i, j, k, 1) +=
672  (W2A_vel(i, j, k, 1) - vel(i, j, k, 1)) *
673  (newtime - t_last) / (W2A_t - t_last + 1e-16);
674  vel(i, j, k, 2) +=
675  (W2A_vel(i, j, k, 2) - vel(i, j, k, 2)) *
676  (newtime - t_last) / (W2A_t - t_last + 1e-16);
677  });
678  }
679  }
680 #else
681  amrex::ignore_unused(data);
682 #endif
683  }
